# Build from source

```
$ mkdir $PWD/_install
$ ./configure \
    --prefix=$PWD/_install \
    --with-features=huge \
    --enable-gpm \
    --enable-acl \
    --with-x=no \
    --disable-gui \
    --enable-multibyte \
    --enable-python3interp=dynamic CFLAGS='-g'
$ (edit src/Makefile to add STRIP = /bin/true, so that debug symbol won't go away)
$ make install
$ ldd ./_install/bin/vim
```
