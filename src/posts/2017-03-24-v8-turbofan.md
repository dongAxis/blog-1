# Example

I experimented v8 with this classic example:

```
// fib.js

function fib0(n) {
  if (n <= 1) {
    return n;
  } else {
    return fib0(n - 1) + fib0(n - 2);
  }
}

function fib1(n) {
  var arr = [0, 1];
  for(var i = 2; i <= n; i++) {
    arr.push(arr[i - 1] + arr[i - 2]);
  }
  return arr[n];
}

function main() {
  print(fib0(10)); // => 55
  print(fib1(10)); // => 55
}

main();
```
