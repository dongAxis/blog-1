$ cat ~/.bashrc

function de() {
    case $1 in
        gnome-x  ) startx gnome-session ;;
        gnome-wl ) XDG_SESSION_TYPE=wayland exec dbus-run-session gnome-session ;;
        sway     ) XDG_SESSION_TYPE=wayland \
                   XKB_DEFAULT_OPTIONS=ctrl:nocaps \
                   exec dbus-run-session sway ;;
    esac
}

$ de sway
